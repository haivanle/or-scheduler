Google Apps Script — Write-back & Email

To write confirmed bookings back to your sheet and send email reports, deploy a Google Apps Script Web App. Follow the steps below once, then paste the URL here.

How to set up the Apps Script (click to expand)
  1. Open your Google Sheet → Extensions → Apps Script
  2. Delete any existing code and paste the Apps Script code below
  3. Click Deploy → New deployment
  4. Type: Web app · Execute as: Me · Who has access: Anyone
  5. Click Deploy, authorise when prompted, copy the Web App URL
  6. Paste the URL in the field below
function doPost(e) { try { var payload = JSON.parse(e.postData.contents); // ── Security: reject requests without the correct secret token ── var SECRET_TOKEN = 'YOUR_SECRET_TOKEN_HERE'; // change this to your own secret if (payload.token !== SECRET_TOKEN) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: 'Unauthorized' })) .setMimeType(ContentService.MimeType.JSON); } var action = payload.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'writeBookings') { var tabName = 'Confirmed Bookings'; var sheet = ss.getSheetByName(tabName) || ss.insertSheet(tabName); sheet.clearContents(); var headers = ['Booking ID','Procedure','Surgeon','Consult Date', 'Surgery Date','OR Room','Start Time','End Time', 'Duration (min)','Revenue','Confirmed At']; sheet.appendRow(headers); payload.bookings.forEach(function(b) { sheet.appendRow(b); }); // Auto-resize columns sheet.autoResizeColumns(1, headers.length); } if (action === 'sendEmail') { var to = payload.to; var subject = payload.subject; var body = payload.body; MailApp.sendEmail({ to: to, subject: subject, htmlBody: body }); } return ContentService.createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } }
